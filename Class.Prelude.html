module _ {a b c} where
  3REL : (A : Set a) (B : Set b) (C : Set c) (ℓ : Level) → Type _
  3REL A B C ℓ = A → B → C → Type ℓ

  Decidable³ : ∀ {A B C ℓ} → 3REL A B C ℓ → Type _
  Decidable³ _~_~_ = ∀ x y z → Dec (x ~ y ~ z)
